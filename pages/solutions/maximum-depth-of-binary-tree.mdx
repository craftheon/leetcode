```javascript
/**
 * @param {TreeNode} root
 * @return {number}
 */
var maxDepth = function (root) {
  let depth = 0, res = 0
  const traverse = function (root) {
    if (!root) return null
    depth++
    if (root.left == null && root.right == null) { // tree bottom
      res = Math.max(res, depth);
    }
    traverse(root.left)
    traverse(root.right)
    depth--
  }
  traverse(root)
  return res
};
```
