## Solution


```javascript
function twoSum(nums, target) {
  const map = {}
  for (const [index, num] of nums.entries()) {
    if (map[target - num] !== undefined) {
      return [map[target - num], index]
    }
    map[num] = index
  }
  return []
}
```
