```javascript
/**
 * @param {TreeNode} root
 * @return {boolean}
 */
var isBalanced = function(root) {
  let res = true
  var traverse = function(root) {
    if (!root) return 0
    let l = traverse(root.left),
      r = traverse(root.right)
    if (Math.abs(l - r) > 1) {
      res = false
    }
    return Math.max(l, r) + 1
  }
  traverse(root)
  return res
};
```
